БД представляет собой бэкенд для "Офиса". Реализуется функционал занесения/хранения основной важной информации для
работы офиса. В последствии планирую добавить визуальный интерфейс и несколько функция для уведмолений в телеграм или почту.

Основные таблицы.
empl - Основная таблица Сотрудники.
id varchar(50) --системный айди,
dadd datetime  --дата добавления в систему,
[name] varchar(150) --имя сотрудника,
sex varchar(1) 	    --пол.,
birthday date       --день рождения,
dstatr date	    --Дата устройства,
dend date	    --Дата увольнения,
fw varchar(1)       --Признак первой работы.


empl_param -Связующая таблица.Параметры сотрудника, его привязка к отделу, технике и т.д.
id varchar(50)	--системный айди,
dadd datetime	--дата добавления в систему,
empl_id varchar(50) --айди сотрудника,
[type] varchar(100) --тип связи,
[value] varchar(50) --значение для типа свящи,
a varchar(1)        --актуальна ли запись в данный момент(активна или нет),
dend datetime       --дата окончания актуальности


department 	--таблица справочник, отделы в системе
id varchar(50)	--системный айди.,
[name] varchar(100) --имя отдела,
a varchar(1)	--активность/актуальность

 
position	--таблица справочник, должность
id varchar(50)   --системный айди,
[name] varchar(100) --название должности,
a varchar(1)	   --активность/актуальность


equipment 	--таблица, хранящая информацию о технике.
id varchar(50)  --системный айди,
dadd datetime   --дата добавления,
[type] varchar(100) --тип техники(ноутбук,мышка и тд),
sn varchar(100)     --учетный номер техники,
account varchar(100) --аккаунт,
[name] varchar(100)  --название,
[model] varchar(100) --модель,
collor varchar(100)  --цвет,
a varchar(1)         --активность,
dend datetime        --дата прекращения


event_dc 	--таблица справочник, для событий(напр. командировка, больничный)
id varchar(50)  --системный айди,
dadd datetime   --дата добавлениия,
[name] varchar(100) --краткое название,
[desc] varchar(500) --Описание


[event]		--таблица привязки события/сотрудника
id varchar(50) 	--системный айди,
dadd datetime	--дата добавления записи,
event_id varchar(50) --айди события из справочника,
empl_id varchar(50)  --айди сотрудника,
dstart date	     --дата начала события,
dend date	     --дата окончания

vacation      	--таблица отпусков
id varchar(50)  --системный айди,
dadd datetime	--дата добавления запис,
empl_id varchar(50) --айди сотрудника,
vacstart date	    --дата начала,
vacend date	    --дата окончания,	
a varchar(1)	    --актуальность



wifi		--таблица справочник для вайфая
id varchar(50)  --системный айди,
dadd datetime   --дата добавления записи,
[name] varchar(100)  --название,
pass varchar(100)    --пароль,
a varchar(1) 	     --актуальность



workmode 		--таблица режима работыы
id varchar(50)		--системный айди
dadd datetime		--дата добавления,
empl_id varchar(50)	--айди сотрудника,
monday varchar(50)	--ДЕнЬ,
tuesday varchar(50),
wednesday varchar(50),
thursday varchar(50),
friday varchar(50),
saturday varchar(50),
sunday varchar(50),
dinner varchar(50)	--обед,
a varchar(1)		--актуальность.
